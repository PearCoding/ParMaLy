%token NUM;
%start in;

in:	add_expr
	;

add_expr:	mul_expr
			| mul_expr '+' add_expr
			| mul_expr '-' add_expr
			;
			
mul_expr:	unary_expr
			| unary_expr '*' mul_expr
			| unary_expr '/' mul_expr
			| unary_expr '%' mul_expr
			| unary_expr '**' mul_expr
			;
			
unary_expr:	'(' add_expr ')'
			| literal
			;
			
literal:	NUM
			;
